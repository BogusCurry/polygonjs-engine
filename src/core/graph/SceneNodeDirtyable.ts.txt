export class SceneNodeDirtyable extends GraphNode(Dirtyable(CoreObject)) {

	}

	set_scene(scene: Scene): void {
		throw 'SceneNodeDirtyable.set_scene requires implementation'
	}

		return this._scene
	}

	}

		return this.graph_successors()
	}
